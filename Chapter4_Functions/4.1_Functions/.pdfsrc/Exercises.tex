\begin{enumerate}
\item Give the output of the following programs:
\begin{enumerate}
\item 
\begin{lstlisting}[style=CStyle]
#include <stdio.h>

int twice(int a) {
    return a * 2;
}

int main(void) {
    printf("%d\n", 4 * twice(3));
    return 0;
}
\end{lstlisting}

\item
\begin{lstlisting}[style=CStyle]
#include <stdio.h>

int half_and_decrement(int num) {
    num = num / 2;
    printf("%d\n", num);
    num = num - 1;
    return num;
}

int main(void) {
    int a = 70;
    printf("%d\n", half_and_decrement(a));
    printf("%d\n", a);
    a = half_and_decrement(a);
    printf("%d\n", a);
    return 0;
}
\end{lstlisting}

\item
\begin{lstlisting}[style=CStyle]
#include <stdio.h>

double average(double a, double b) {
    a = a / 2;
    printf("%d\n", a);
    b = b / 2;
    printf("%d\n", b);
    return a + b;
}

int main(void) {
    double a = 3.14;
    double b = 2.72;
    printf("%f\n", average(a, b));
    printf("%f\n", a);
    printf("%f\n", b);
    return 0;
}

\end{lstlisting}
\end{enumerate}

\item

\end{enumerate}
